int Nparticles = 1000, index;
double arrayX[1000];
double arrayY[1000];
double xj[1000];
double yj[1000];
double CDF[1000];
double u[1000];

####Annotation Begins####
for(j = 0; j<Nparticles; j++)
{
   
   int index = -1;
   
   for(int x = 0; x < Nparticles; x++) //FRATIO1000
   {
      if(CDF[x] >= u[j])	//taken very rarely// BRATIO0.0001
      {
         index = x;
         break;
      }
   }
 
   if(index == -1)		//never taken // BRATIO0.0
   {
      i = Nparticles - 1;
   }
   else  //BRATIO0.0
   {
      i = index;
   }
   
   if(i == -1)			//never taken // BRATIO0.0
   { 
      i = Nparticles-1;
   }
   
   xj[j] = arrayX[i];
   yj[j] = arrayY[i];
   
}
